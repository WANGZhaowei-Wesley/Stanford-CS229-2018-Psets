\begin{align*}
    P(y;\lambda) &= \frac{e^{-\lambda}\lambda^y}{y!}\\
    &= \frac{1}{y!}e^{y log\lambda-\lambda}\\
\end{align*}
It's easy to see that:
\begin{align*}
    &b(y) = \frac{1}{y!}\\
    &\eta = log\lambda\\
    &T(y) = y\\
    &a(\eta) = \lambda = e^\eta\\
\end{align*}
\end{answer}
